(defmacro define-actor ((var context) &body body)
  (let ((start-fn (gensym "START-FN-")))

    `(let ((,start-fn (lambda ()
                        (setf ,var (actor-of ,context
                                             :name (symbol-name ,var)
                                             :receive ,@body)))))
       (nhooks:add-hook star:*actors-start-hook* ,start-fn :append t))))
